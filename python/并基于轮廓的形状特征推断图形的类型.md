以下脚本实现了**从图像四周向中心发射射线**的方式遍历像素，利用透明度和颜色差异判断背景与前景，从而提取图形的外轮廓，并基于轮廓的形状特征（如距离方差、角点数量）推断图形的类型（圆形、矩形、三角形等）。

```python
#!/usr/bin/env python3
"""
从四周向中心遍历像素，根据颜色/透明度变化检测图形形状。

原理：
  1. 将图像中心作为目标点。
  2. 从图像的每一边界像素向中心点生成直线（Bresenham算法）。
  3. 沿直线逐个像素检测，找到第一个“非背景”像素作为该方向上的形状轮廓点。
  4. 收集所有轮廓点后，分析其到中心的距离变化，判断形状类型。
  5. 支持透明度优先的背景判断，也可用颜色距离（如纯色背景）。
"""

import argparse
import sys
import math
import numpy as np
from PIL import Image, ImageDraw


def bresenham_line(x0, y0, x1, y1):
    """生成从 (x0,y0) 到 (x1,y1) 直线上的所有像素坐标（含两端）"""
    points = []
    dx = abs(x1 - x0)
    dy = abs(y1 - y0)
    sx = 1 if x0 < x1 else -1
    sy = 1 if y0 < y1 else -1
    err = dx - dy
    x, y = x0, y0
    while True:
        points.append((x, y))
        if x == x1 and y == y1:
            break
        e2 = 2 * err
        if e2 > -dy:
            err -= dy
            x += sx
        if e2 < dx:
            err += dx
            y += sy
    return points


def detect_shape_radial(image, background_color=None, alpha_thresh=128, color_thresh=30.0):
    """
    通过从图像边界向中心发射射线的方式提取形状轮廓点。

    参数:
        image: PIL.Image, 输入图像
        background_color: tuple(R,G,B) 或 None。None时自动检测背景。
        alpha_thresh: 透明度阈值，alpha < 此值视为背景（0-255）
        color_thresh: 颜色距离阈值，小于此值视为背景

    返回:
        contour_points: [(x,y), ...] 轮廓点序列（按边界环绕顺序）
        center: (cx, cy) 图像中心
        use_alpha: 是否使用alpha通道进行背景判定
    """
    img = image.convert("RGBA")
    w, h = img.size
    pixels = np.array(img)

    # 图像中心（几何中心）
    cx = (w - 1) / 2.0
    cy = (h - 1) / 2.0

    # ----- 判定背景依据 -----
    use_alpha = False
    if background_color is None:
        # 检查图像中是否存在透明像素
        min_alpha = np.min(pixels[:, :, 3])
        if min_alpha < alpha_thresh:
            use_alpha = True
        else:
            # 无透明区域，自动取四角平均颜色作为背景色
            corners = [(0, 0), (w-1, 0), (0, h-1), (w-1, h-1)]
            cols = [pixels[y, x, :3] for x, y in corners]
            background_color = tuple(int(c) for c in np.mean(cols, axis=0))
            print(f"自动检测背景色: {background_color}")

    # ----- 构建边界像素序列（顺时针环绕） -----
    boundary = []
    # 上边（左→右）
    for x in range(w):
        boundary.append((x, 0))
    # 右边（上→下）
    for y in range(1, h):
        boundary.append((w - 1, y))
    # 下边（右→左）
    for x in range(w - 2, -1, -1):
        boundary.append((x, h - 1))
    # 左边（下→上）
    for y in range(h - 2, 0, -1):
        boundary.append((0, y))

    # ----- 遍历每条射线，找第一个前景点 -----
    contour_points = []
    for bx, by in boundary:
        line = bresenham_line(bx, by, int(round(cx)), int(round(cy)))
        hit = None
        for x, y in line:
            if 0 <= x < w and 0 <= y < h:
                r, g, b, a = pixels[y, x]
                if use_alpha:
                    if a >= alpha_thresh:          # 遇到不透明像素
                        hit = (x, y)
                        break
                else:
                    if a < alpha_thresh:
                        continue                    # 透明像素直接视为背景
                    br, bg, bb = background_color
                    dist = math.sqrt((r - br) ** 2 + (g - bg) ** 2 + (b - bb) ** 2)
                    if dist >= color_thresh:        # 颜色差异足够大
                        hit = (x, y)
                        break
        # 若整条射线都为背景，则记录中心点（或 None），这里用中心点填充
        if hit is None:
            hit = (int(round(cx)), int(round(cy)))
        contour_points.append(hit)

    return contour_points, (cx, cy), use_alpha


def analyze_shape(contour_points, center):
    """
    根据轮廓点到中心的距离变化规律判断形状类型。

    返回:
        形状名称字符串: 圆形/正方形/矩形/三角形/多边形(角数)/未知
    """
    cx, cy = center
    distances = []
    angles = []
    for x, y in contour_points:
        dx = x - cx
        dy = y - cy
        distances.append(math.hypot(dx, dy))
        angles.append(math.atan2(dy, dx))

    if not distances:
        return "未知"

    dist_arr = np.array(distances)
    mean_dist = np.mean(dist_arr)
    std_dist = np.std(dist_arr)

    # ① 圆形：距离波动极小
    if mean_dist > 0 and std_dist / mean_dist < 0.1:
        return "圆形"

    # ② 根据角点数量判断多边形
    # 按角度排序后统计局部极大值（角点）
    sorted_idx = np.argsort(angles)
    sorted_dist = dist_arr[sorted_idx]
    n = len(sorted_dist)
    peaks = 0
    for i in range(n):
        prev = sorted_dist[(i - 1) % n]
        curr = sorted_dist[i]
        nxt = sorted_dist[(i + 1) % n]
        if curr > prev and curr > nxt:
            peaks += 1

    if peaks == 3:
        return "三角形"
    elif peaks == 4:
        # 进一步区分正方形/矩形：最大与最小距离之比
        ratio = np.max(dist_arr) / np.min(dist_arr) if np.min(dist_arr) > 0 else 999
        if 1.0 <= ratio < 1.5:
            return "正方形"
        else:
            return "矩形"
    elif peaks == 0:
        return "圆形"   # 无峰值说明轮廓平滑
    elif peaks > 4:
        return f"多边形({peaks}个角)"
    else:
        return "未知"


def main():
    parser = argparse.ArgumentParser(
        description="从四周向中心遍历像素，基于颜色/透明度变化检测图形形状"
    )
    parser.add_argument("input", help="输入图像路径")
    parser.add_argument("-o", "--output", help="输出图像路径（可视化轮廓）")
    parser.add_argument("-b", "--background", 
                        help="背景颜色，格式 R,G,B（如 255,255,255）。不指定则自动检测。")
    parser.add_argument("-a", "--alpha-thresh", type=int, default=128,
                        help="透明度阈值(0-255)，alpha小于此值视为背景 (默认: 128)")
    parser.add_argument("-c", "--color-thresh", type=float, default=30.0,
                        help="颜色距离阈值，小于此值视为背景 (默认: 30)")
    args = parser.parse_args()

    # 解析背景颜色
    bg_color = None
    if args.background:
        parts = args.background.split(",")
        if len(parts) == 3:
            bg_color = tuple(int(p.strip()) for p in parts)
        else:
            print("错误：背景颜色格式应为 R,G,B")
            sys.exit(1)

    # 加载图像
    try:
        img = Image.open(args.input)
    except Exception as e:
        print(f"无法打开图像: {e}")
        sys.exit(1)

    # 检测形状轮廓
    contour, center, use_alpha = detect_shape_radial(
        img, bg_color, args.alpha_thresh, args.color_thresh
    )

    # 分析形状
    shape_type = analyze_shape(contour, center)
    print(f"检测到的形状: {shape_type}")
    print(f"轮廓点数量: {len(contour)}")

    # 可选的可视化输出
    if args.output:
        out_img = img.convert("RGB")
        draw = ImageDraw.Draw(out_img)

        # 按角度排序轮廓点并连线，显示检测到的外轮廓
        cx, cy = center
        angled = []
        for x, y in contour:
            angle = math.atan2(y - cy, x - cx)
            angled.append((angle, x, y))
        angled.sort(key=lambda t: t[0])
        sorted_pts = [(x, y) for _, x, y in angled]

        if len(sorted_pts) > 2:
            draw.line(sorted_pts + [sorted_pts[0]], fill=(255, 0, 0), width=2)

        # 标记中心点
        r = 3
        draw.ellipse((cx - r, cy - r, cx + r, cy + r), fill=(0, 255, 0))

        out_img.save(args.output)
        print(f"轮廓图已保存到 {args.output}")


if __name__ == "__main__":
    main()
```

## 使用方法

1. **安装依赖**（需要 `Pillow` 和 `numpy`）：
   ```bash
   pip install Pillow numpy
   ```

2. **基本运行**：
   ```bash
   python shape_detector.py input.png
   ```
   脚本会输出检测到的形状类型（如“圆形”、“正方形”等）。

3. **保存可视化结果**：
   ```bash
   python shape_detector.py input.png -o output.png
   ```
   生成一张在原图上标出红色外轮廓和绿色中心点的图像。

4. **手动指定背景颜色**（例如纯白背景）：
   ```bash
   python shape_detector.py photo.jpg -b 255,255,255
   ```

5. **调整判定阈值**（适用于边缘模糊或噪点较多的图）：
   ```bash
   python shape_detector.py input.png -a 200 -c 50
   ```
   - `-a`：透明度阈值，低于此值视为背景（对 PNG 透明区域有效）。
   - `-c`：颜色差异阈值，像素与背景色距离小于此值视为背景。

## 工作原理简析

- **从四周向中心遍历**：将图像边界上的每一个像素作为起点，向图像几何中心连一条直线（射线），逐像素检查颜色/透明度。
- **前景判定**：
  - 优先使用 Alpha 通道：透明度 ≥ 阈值即为前景。
  - 若无透明通道或全图不透明，自动取图像四角平均色作为背景，通过颜色距离区分前景。
- **轮廓提取**：每条射线第一个命中前景的像素被记下，所有射线完成后得到一组环绕形状的轮廓点。
- **形状识别**：计算这些点到中心的距离，若距离几乎恒定则为**圆形**；若出现 3 个局部极大值则为**三角形**；4 个极值且距离比例接近 1:1 为**正方形**，比例更大则为**矩形**；其余为多边形或未知。
