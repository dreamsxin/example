# 缓冲区溢出

```c
#include <stdio.h>
#include <stdlib.h>

//修改函数回调的返回地址
void foo(void) {
        int a, *p;
        p = (int*)((char*)&a + 12); //12字节 (int*)(&a + 3)
        *p += 12;
}

int main(void) {
        foo();
        printf("我是正常的返回地址");
        printf("我是被修改的返回地址");
}
```
```shell
gcc -g -z execstack -fno-stack-protector test.c
```
