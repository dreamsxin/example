```python
"""数据模型定义。"""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from typing import Any, Literal

Role = Literal["system", "user", "assistant"]


@dataclass
class Message:
    """对话消息。"""

    role: Role
    content: str
    timestamp: datetime = field(default_factory=datetime.now)
    metadata: dict[str, Any] = field(default_factory=dict)

    def to_dict(self) -> dict[str, str]:
        return {"role": self.role, "content": self.content}

    def __len__(self) -> int:
        return len(self.content)


@dataclass
class CompressionResult:
    """压缩结果，包含统计信息。"""

    messages: list[Message]
    original_count: int
    compressed_count: int
    original_tokens: int
    compressed_tokens: int
    strategy_name: str

    @property
    def compression_ratio(self) -> float:
        if self.original_tokens == 0:
            return 0.0
        return 1 - (self.compressed_tokens / self.original_tokens)

    def summary_line(self) -> str:
        return (
            f"[{self.strategy_name}] "
            f"消息 {self.original_count}→{self.compressed_count} 条 | "
            f"Token {self.original_tokens}→{self.compressed_tokens} "
            f"(压缩 {self.compression_ratio:.0%})"
        )
```

```python
"""Token 计数工具，支持精确计数 (tiktoken) 和回退估算。"""

from __future__ import annotations

from models import Message

# 尝试加载 tiktoken，失败则使用估算
try:
    import tiktoken

    _encoder = tiktoken.get_encoding("cl100k_base")

    def count_tokens(text: str) -> int:
        """使用 tiktoken 精确计数。"""
        return len(_encoder.encode(text))

    TIKTOKEN_AVAILABLE = True

except ImportError:
    TIKTOKEN_AVAILABLE = False

    def count_tokens(text: str) -> int:
        """回退估算：英文 ~4 字符/token，中文 ~1.5 字符/token。"""
        ascii_chars = sum(1 for c in text if ord(c) < 128)
        non_ascii = len(text) - ascii_chars
        return max(1, (ascii_chars // 4) + int(non_ascii / 1.5))


def count_message_tokens(message: Message) -> int:
    """计算单条消息的 token 数（含 role 开销）。"""
    # 每条消息约有 4 token 的格式开销
    return count_tokens(message.content) + count_tokens(message.role) + 4


def count_messages_tokens(messages: list[Message]) -> int:
    """计算消息列表的总 token 数。"""
    return sum(count_message_tokens(m) for m in messages)
```

```python
        temperature: float = 0.7,
        max_tokens: int = 1024,
        **kwargs: Any,
    ) -> str:
        """发送对话请求，返回助手回复文本。"""
        ...


class OpenAIClient(LLMClient):
    """OpenAI 兼容客户端（支持 OpenAI / DeepSeek / 本地部署等）。"""

    def __init__(
        self,
        api_key: str,
        model: str = "gpt-4o-mini",
        base_url: str | None = None,
    ):
        from openai import OpenAI

        self.client = OpenAI(api_key=api_key, base_url=base_url)
        self.model = model

    def chat(self, messages, temperature=0.7, max_tokens=1024, **kwargs):
        response = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            temperature=temperature,
            max_tokens=max_tokens,
            **kwargs,
        )
        return response.choices[0].message.content


class EchoClient(LLMClient):
    """回显客户端，用于开发调试。"""

    def chat(self, messages, temperature=0.7, max_tokens=1024, **kwargs):
        last = messages[-1] if messages else {}
        content = last.get("content", "")
        preview = content[:80] + ("..." if len(content) > 80 else "")
        return f"[Echo] 收到 {len(messages)} 条消息。最后一条: {preview}"
```

```python
"""压缩策略抽象基类。"""

from __future__ import annotations

from abc import ABC, abstractmethod
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from models import Message


class CompressionStrategy(ABC):
    """所有压缩策略的基类。"""

    @property
    @abstractmethod
    def name(self) -> str:
        """策略名称，用于展示。"""
        ...

    @property
    @abstractmethod
    def description(self) -> str:
        """策略描述。"""
        ...

    @abstractmethod
    def compress(self, messages: list[Message]) -> list[Message]:
        """
        压缩消息列表。

        Args:
            messages: 完整的对话历史。

        Returns:
            压缩后的消息列表，可直接发送给 LLM。
        """
        ...

    def _split_system_messages(
        self, messages: list[Message]
    ) -> tuple[list[Message], list[Message]]:
        """分离 system 消息和非 system 消息。"""
        system_msgs = [m for m in messages if m.role == "system"]
        other_msgs = [m for m in messages if m.role != "system"]
        return system_msgs, other_msgs
```

```python
"""滑动窗口策略 — 保留最近 N 条消息。"""

from __future__ import annotations

from compression.base import CompressionStrategy
from models import Message


class SlidingWindow(CompressionStrategy):
    """
    保留最近的 N 条非 system 消息，system 消息始终保留。

    适用场景：对话很长但只需要近期上下文。
    """

    def __init__(self, window_size: int = 10):
        if window_size < 1:
            raise ValueError("window_size 必须 >= 1")
        self.window_size = window_size

    @property
    def name(self) -> str:
        return "sliding_window"

    @property
    def description(self) -> str:
        return f"滑动窗口：保留最近 {self.window_size} 条消息"

    def compress(self, messages: list[Message]) -> list[Message]:
        system_msgs, other_msgs = self._split_system_messages(messages)

        if len(other_msgs) <= self.window_size:
            return list(messages)

        kept = other_msgs[-self.window_size :]
        return system_msgs + kept
```

### `compression/token_budget.py`

```python
"""Token 预算策略 — 在 token 限制内尽可能保留更多近期消息。"""

from __future__ import annotations

from compression.base import CompressionStrategy
from models import Message
from token_utils import count_message_tokens


class TokenBudget(CompressionStrategy):
    """
    设定 token 预算上限，从最新消息向前回溯，尽可能多地保留消息。

    适用场景：需要精确控制发送给 LLM 的 token 数量。
    """

    def __init__(self, max_tokens: int = 3000, reserve_for_response: int = 500):
        if max_tokens < 100:
            raise ValueError("max_tokens 必须 >= 100")
        self.max_tokens = max_tokens
        self.reserve_for_response = reserve_for_response

    @property
    def name(self) -> str:
        return "token_budget"

    @property
    def description(self) -> str:
        return f"Token 预算：最多 {self.max_tokens} tokens"

    def compress(self, messages: list[Message]) -> list[Message]:
        system_msgs, other_msgs = self._split_system_messages(messages)

        # system 消息占用的 token 必须扣除
        budget = self.max_tokens - sum(count_message_tokens(m) for m in system_msgs)

        if budget <= 0:
            # system 消息已超预算，只保留 system 消息
            return system_msgs

        # 从最新消息向前填充
        selected: list[Message] = []
        used = 0
        for msg in reversed(other_msgs):
            cost = count_message_tokens(msg)
            if used + cost > budget:
                break
            selected.append(msg)
            used += cost

        selected.reverse()
        return system_msgs + selected
```

### `compression/summary.py`

```python
"""摘要压缩策略 — 使用 LLM 将旧消息压缩为摘要。"""

from __future__ import annotations

from typing import Callable

from compression.base import CompressionStrategy
from models import Message


class SummaryCompression(CompressionStrategy):
    """
    将较早的消息用 LLM 生成摘要，最近的消息保留原文。

    适用场景：需要保留关键上下文但对话过长。
    """

    def __init__(
        self,
        llm_caller: Callable[[list[dict[str, str]]], str],
        keep_recent: int = 6,
        summary_instruction: str | None = None,
    ):
        """
        Args:
            llm_caller: 调用 LLM 的函数，签名 (messages) -> str。
            keep_recent: 保留最近 N 条消息的原文。
            summary_instruction: 自定义摘要指令。
        """
        self.llm_caller = llm_caller
        self.keep_recent = keep_recent
        self.summary_instruction = summary_instruction or (
            "请将以下对话压缩为简洁的摘要，保留：\n"
            "1. 关键事实和决策\n"
            "2. 重要的上下文信息\n"
            "3. 未完成的任务或待解决的问题\n"
            "使用第三人称客观叙述，不要遗漏重要细节。"
        )

    @property
    def name(self) -> str:
        return "summary"

    @property
    def description(self) -> str:
        return f"LLM 摘要：旧消息压缩为摘要，保留最近 {self.keep_recent} 条原文"

    def compress(self, messages: list[Message]) -> list[Message]:
        system_msgs, other_msgs = self._split_system_messages(messages)

        if len(other_msgs) <= self.keep_recent:
            return list(messages)

        to_summarize = other_msgs[: -self.keep_recent]
        recent = other_msgs[-self.keep_recent :]

        # 构造摘要请求
        conversation_text = "\n".join(
            f"[{m.role}] {m.content}" for m in to_summarize
        )
        prompt_messages = [
            {"role": "system", "content": self.summary_instruction},
            {"role": "user", "content": conversation_text},
        ]

        try:
            summary_text = self.llm_caller(prompt_messages)
        except Exception as e:
            # LLM 调用失败时回退到截断
            summary_text = (
                f"[摘要生成失败: {e}]\n"
                f"原始对话包含 {len(to_summarize)} 条消息，已被截断。"
            )

        summary_msg = Message(
            role="system",
            content=f"[对话摘要]\n{summary_text}",
            metadata={"compressed_from": len(to_summarize)},
        )

        return system_msgs + [summary_msg] + recent
```

### `compression/smart.py`

```python
"""分层智能压缩策略 — 三级压缩，逐层降低信息密度。"""

from __future__ import annotations

from typing import Callable

from compression.base import CompressionStrategy
from models import Message


class SmartCompression(CompressionStrategy):
    """
    三层压缩：
      - 最近的消息：保留原文
      - 中间层：生成详细摘要
      - 最早的消息：提取关键事实列表

    适用场景：超长对话，需要最大化保留有效信息。
    """

    def __init__(
        self,
        llm_caller: Callable[[list[dict[str, str]]], str],
        keep_recent: int = 6,
        mid_ratio: float = 0.5,
    ):
        """
        Args:
            llm_caller: LLM 调用函数。
            keep_recent: 最近 N 条保留原文。
            mid_ratio: 中间层在剩余消息中的占比 (0-1)。
        """
        self.llm_caller = llm_caller
        self.keep_recent = keep_recent
        self.mid_ratio = mid_ratio

    @property
    def name(self) -> str:
        return "smart"

    @property
    def description(self) -> str:
        return (
            f"分层压缩：最近 {self.keep_recent} 条原文 + "
            f"中层详细摘要 + 远层关键事实"
        )

    def compress(self, messages: list[Message]) -> list[Message]:
        system_msgs, other_msgs = self._split_system_messages(messages)

        if len(other_msgs) <= self.keep_recent:
            return list(messages)

        old_msgs = other_msgs[: -self.keep_recent]
        recent = other_msgs[-self.keep_recent :]

        # 将旧消息分为 mid 和 far 两层
        split_point = int(len(old_msgs) * self.mid_ratio)
        mid_msgs = old_msgs[split_point:]
        far_msgs = old_msgs[:split_point]

        result = list(system_msgs)

        # 远层：提取关键事实
        if far_msgs:
            facts = self._extract_facts(far_msgs)
            result.append(
                Message(
                    role="system",
                    content=f"[关键事实提取]\n{facts}",
                    metadata={"layer": "far", "compressed_from": len(far_msgs)},
                )
            )

        # 中层：详细摘要
        if mid_msgs:
            summary = self._summarize(mid_msgs)
            result.append(
                Message(
                    role="system",
                    content=f"[对话摘要]\n{summary}",
                    metadata={"layer": "mid", "compressed_from": len(mid_msgs)},
                )
            )

        result.extend(recent)
        return result

    def _extract_facts(self, messages: list[Message]) -> str:
        text = "\n".join(f"[{m.role}] {m.content}" for m in messages)
        prompt = [
            {
                "role": "system",
                "content": (
                    "从以下对话中提取关键事实，以简洁的要点列表输出。\n"
                    "每个要点一行，以 '•' 开头。只保留最重要的信息。"
                ),
            },
            {"role": "user", "content": text},
        ]
        try:
            return self.llm_caller(prompt)
        except Exception as e:
            return f"[提取失败: {e}] 共 {len(messages)} 条消息"

    def _summarize(self, messages: list[Message]) -> str:
        text = "\n".join(f"[{m.role}] {m.content}" for m in messages)
        prompt = [
            {
                "role": "system",
                "content": (
                    "将以下对话压缩为简洁摘要，保留关键事实、决策和上下文。"
                ),
            },
            {"role": "user", "content": text},
        ]
        try:
            return self.llm_caller(prompt)
        except Exception as e:
            return f"[摘要失败: {e}] 共 {len(messages)} 条消息"
```

```python
"""压缩策略注册表。"""

from __future__ import annotations

from typing import Callable

from compression.base import CompressionStrategy
from compression.no_compression import NoCompression
from compression.sliding_window import SlidingWindow
from compression.smart import SmartCompression
from compression.summary import SummaryCompression
from compression.token_budget import TokenBudget

# 策略注册表：名称 → 需要 LLM 的标记
STRATEGY_REGISTRY: dict[str, bool] = {
    "none": False,
    "sliding_window": False,
    "token_budget": False,
    "summary": True,
    "smart": True,
}

STRATEGY_DESCRIPTIONS = {
    "none": "不压缩 — 保留完整对话历史",
    "sliding_window": "滑动窗口 — 保留最近 N 条消息",
    "token_budget": "Token 预算 — 在 token 限制内保留尽可能多的消息",
    "summary": "LLM 摘要 — 旧消息压缩为摘要，近期消息保留原文",
    "smart": "分层压缩 — 三级信息密度（原文/摘要/关键事实）",
}


def create_strategy(
    name: str,
    llm_caller: Callable | None = None,
    **kwargs,
) -> CompressionStrategy:
    """
    工厂函数：根据名称创建压缩策略实例。

    Args:
        name: 策略名称 (none / sliding_window / token_budget / summary / smart)。
        llm_caller: LLM 调用函数（summary 和 smart 策略需要）。
        **kwargs: 传递给策略构造函数的额外参数。
    """
    if name not in STRATEGY_REGISTRY:
        raise ValueError(
            f"未知策略 '{name}'，可选: {list(STRATEGY_REGISTRY.keys())}"
        )

    if STRATEGY_REGISTRY[name] and llm_caller is None:
        raise ValueError(f"策略 '{name}' 需要提供 llm_caller")

    match name:
        case "none":
            return NoCompression()
        case "sliding_window":
            return SlidingWindow(**kwargs)
        case "token_budget":
            return TokenBudget(**kwargs)
        case "summary":
            return SummaryCompression(llm_caller=llm_caller, **kwargs)
        case "smart":
            return SmartCompression(llm_caller=llm_caller, **kwargs)
        case _:
            raise ValueError(f"未实现的策略: {name}")


__all__ = [
    "CompressionStrategy",
    "NoCompression",
    "SlidingWindow",
    "TokenBudget",
    "SummaryCompression",
    "SmartCompression",
    "create_strategy",
    "STRATEGY_REGISTRY",
    "STRATEGY_DESCRIPTIONS",
]
```

### `agent.py`

```python
"""Agent 核心 — 编排对话、压缩、调用 LLM。"""

from __future__ import annotations

from typing import Any

from compression import CompressionStrategy, create_strategy
from llm_client import LLMClient
from models import CompressionResult, Message
from token_utils import count_messages_tokens


class ContextCompressionAgent:
    """
    支持上下文压缩的 AI Agent。

    特性：
    - 管理完整对话历史
    - 发送前按选定策略压缩上下文
    - 可随时切换压缩模式
    - 记录每次压缩的统计信息
    """

    def __init__(
        self,
        llm_client: LLMClient,
        compression_strategy: CompressionStrategy | str = "none",
        system_prompt: str | None = None,
        strategy_kwargs: dict[str, Any] | None = None,
    ):
        self.llm_client = llm_client
        self._history: list[Message] = []
        self._compression_log: list[CompressionResult] = []

        # 初始化 system prompt
        if system_prompt:
            self._history.append(Message(role="system", content=system_prompt))

        # 初始化压缩策略
        if isinstance(compression_strategy, str):
            llm_caller = self._make_llm_caller()
            self._strategy = create_strategy(
                compression_strategy,
                llm_caller=llm_caller,
                **(strategy_kwargs or {}),
            )
        else:
            self._strategy = compression_strategy

    # ── 公开接口 ──────────────────────────────────────────

    @property
    def strategy(self) -> CompressionStrategy:
        return self._strategy

    @property
    def history(self) -> list[Message]:
        return list(self._history)

    @property
    def compression_log(self) -> list[CompressionResult]:
        return list(self._compression_log)

    def set_strategy(
        self, name: str, **kwargs: Any
    ) -> CompressionStrategy:
        """切换压缩策略。"""
        llm_caller = self._make_llm_caller()
        self._strategy = create_strategy(name, llm_caller=llm_caller, **kwargs)
        return self._strategy

    def add_message(self, role: str, content: str) -> Message:
        """手动添加消息到历史。"""
        msg = Message(role=role, content=content)  # type: ignore[arg-type]
        self._history.append(msg)
        return msg

    def chat(self, user_input: str, **llm_kwargs: Any) -> str:
        """
        完整的对话流程：
        1. 添加用户消息
        2. 压缩上下文
        3. 调用 LLM
        4. 保存助手回复
        5. 返回回复
        """
        # 1. 添加用户消息
        self.add_message("user", user_input)

        # 2. 压缩
        compressed = self._compress()

        # 3. 调用 LLM
        llm_messages = [m.to_dict() for m in compressed.messages]
        response_text = self.llm_client.chat(llm_messages, **llm_kwargs)

        # 4. 保存回复
        self.add_message("assistant", response_text)

        return response_text

    def get_context_preview(self) -> CompressionResult:
        """预览当前压缩效果（不实际调用 LLM）。"""
        return self._compress()

    def get_stats_summary(self) -> str:
        """获取压缩历史统计摘要。"""
        if not self._compression_log:
            return "尚无压缩记录。"

        lines = ["═══ 压缩历史 ═══"]
        for i, result in enumerate(self._compression_log, 1):
            lines.append(f"  {i}. {result.summary_line()}")
        return "\n".join(lines)

    def clear_history(self, keep_system: bool = True) -> None:
        """清空对话历史。"""
        if keep_system:
            self._history = [m for m in self._history if m.role == "system"]
        else:
            self._history.clear()
        self._compression_log.clear()

    # ── 内部方法 ──────────────────────────────────────────

    def _compress(self) -> CompressionResult:
        """执行压缩并记录结果。"""
        original = self._history
        compressed = self._strategy.compress(original)

        result = CompressionResult(
            messages=compressed,
            original_count=len(original),
            compressed_count=len(compressed),
            original_tokens=count_messages_tokens(original),
            compressed_tokens=count_messages_tokens(compressed),
            strategy_name=self._strategy.name,
        )

        self._compression_log.append(result)
        return result

    def _make_llm_caller(self):
        """创建 LLM 调用闭包，供压缩策略使用。"""

        def caller(messages: list[dict[str, str]]) -> str:
            return self.llm_client.chat(messages, temperature=0.3, max_tokens=512)

        return caller
```

## 五种压缩策略对比

| 策略 | 原理 | 额外开销 | 适用场景 |
|---|---|---|---|
| **none** | 原样传递 | 无 | 短对话、需要完整上下文 |
| **sliding_window** | 保留最近 N 条 | 无 | 对话长但只需近期内容 |
| **token_budget** | 按 token 预算从新到旧填充 | 无 | 需要精确控制成本 |
| **summary** | LLM 摘要旧消息 | 1 次 LLM 调用 | 需要保留关键上下文 |
| **smart** | 三级分层压缩 | 2 次 LLM 调用 | 超长对话、最大化信息保留 |

## 扩展方式

添加自定义压缩策略只需三步：

```python
# 1. 继承基类
from compression.base import CompressionStrategy

class MyStrategy(CompressionStrategy):
    @property
    def name(self) -> str:
        return "my_strategy"

    @property
    def description(self) -> str:
        return "我的自定义策略"

    def compress(self, messages):
        # 你的压缩逻辑
        ...

# 2. 注册到 __init__.py 的 STRATEGY_REGISTRY
STRATEGY_REGISTRY["my_strategy"] = False  # True 如果需要 LLM

# 3. 在 create_strategy 的 match 中添加分支
```

整个设计遵循**开闭原则**：新增策略不需要修改 Agent 核心代码，只需扩展策略注册表。
