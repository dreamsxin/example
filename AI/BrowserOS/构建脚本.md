## 注意事项

默认会下载到：

```text
D:\work\BrowserOS\packages\browseros\.chromium\src
```

也就是脚本里的默认 Chromium `src` 目录。实际执行 `fetch chromium` 时，会在它的父目录运行，所以 depot_tools 会生成类似：

```text
D:\work\BrowserOS\packages\browseros\.chromium\.gclient
D:\work\BrowserOS\packages\browseros\.chromium\src\
```

指定目录用：

```powershell
python bootstrap_build.py --chromium-src D:\chromium\src --yes
```

如果 `D:\chromium\src` 不存在，脚本会创建 `D:\chromium`，然后在 `D:\chromium` 下面跑：

```powershell
fetch --no-history chromium
```

所以指定的路径最好以 `src` 结尾。

## bootstrap_build.py
```python
#!/usr/bin/env python3
"""One-command local Chromium checkout and BrowserOS development build."""

from __future__ import annotations

import argparse
import os
import shutil
import subprocess
import sys
from pathlib import Path
from typing import Callable, Sequence


Runner = Callable[[list[str], Path, dict[str, str] | None], None]


def repo_root_from_script() -> Path:
    return Path(__file__).resolve().parent


def browseros_root(repo_root: Path) -> Path:
    return repo_root / "packages" / "browseros"


def default_chromium_src(repo_root: Path) -> Path:
    return browseros_root(repo_root) / ".chromium" / "src"


def read_chromium_version(version_file: Path) -> str:
    values: dict[str, str] = {}
    for line in version_file.read_text().splitlines():
        if "=" not in line:
            continue
        key, value = line.split("=", 1)
        values[key.strip()] = value.strip()

    required = ["MAJOR", "MINOR", "BUILD", "PATCH"]
    missing = [key for key in required if key not in values]
    if missing:
        raise ValueError(f"{version_file} is missing: {', '.join(missing)}")

    return ".".join(values[key] for key in required)


def resolve_command(command: str) -> str:
    resolved = shutil.which(command)
    if resolved is None:
        raise RuntimeError(
            f"Required command not found: {command}\n"
            "Install Chromium depot_tools and make sure it is on PATH:\n"
            "  https://chromium.googlesource.com/chromium/src/+/main/docs/"
            "depot_tools.md"
        )
    return resolved


def default_arch() -> str:
    if sys.platform == "darwin":
        import platform

        return "arm64" if platform.machine() == "arm64" else "x64"
    if sys.platform.startswith("linux"):
        import platform

        return "arm64" if platform.machine() in ("aarch64", "arm64") else "x64"
    return "x64"


def build_browseros_command(
    chromium_src: Path,
    arch: str,
    build_type: str,
) -> list[str]:
    return [
        sys.executable,
        "-m",
        "build.browseros",
        "build",
        "--chromium-src",
        str(chromium_src),
        "--arch",
        arch,
        "--build-type",
        build_type,
        "--setup",
        "--prep",
        "--build",
    ]


def run_command(command: list[str], cwd: Path, env: dict[str, str] | None = None) -> None:
    print(f"\n$ {' '.join(command)}")
    subprocess.run(command, cwd=cwd, env=env, check=True)


def parse_args(argv: Sequence[str] | None, repo_root: Path) -> argparse.Namespace:
    parser = argparse.ArgumentParser(
        description="Fetch Chromium if needed and run a local BrowserOS dev build.",
    )
    parser.add_argument(
        "--chromium-src",
        type=Path,
        default=default_chromium_src(repo_root),
        help="Path to Chromium src directory. Default: packages/browseros/.chromium/src",
    )
    parser.add_argument(
        "--arch",
        choices=["x64", "arm64"],
        default=default_arch(),
        help="Target architecture. Default: current machine architecture.",
    )
    parser.add_argument(
        "--build-type",
        choices=["debug", "release"],
        default="debug",
        help="Build type passed to BrowserOS build system. Default: debug.",
    )
    parser.add_argument(
        "--skip-fetch",
        action="store_true",
        help="Do not run fetch chromium when the source directory is missing.",
    )
    parser.add_argument(
        "--skip-build",
        action="store_true",
        help="Fetch and sync only; do not compile.",
    )
    parser.add_argument(
        "--yes",
        action="store_true",
        help="Skip the large download/build confirmation prompt.",
    )
    parser.add_argument(
        "--jobs",
        type=int,
        help="Set BROWSEROS_NINJA_JOBS for the compile step.",
    )
    return parser.parse_args(argv)


def ensure_tools() -> dict[str, str]:
    required = ["git", "fetch", "gclient", "gn", "autoninja"]
    if sys.platform == "win32":
        required = ["git", "fetch", "gclient.bat", "gn.bat", "autoninja.bat"]
    resolved = {}
    for command in required:
        resolved[command.split(".", 1)[0]] = resolve_command(command)
    return resolved


def confirm_large_build(args: argparse.Namespace, version: str) -> None:
    if args.yes:
        return

    print("This will download/sync Chromium and build BrowserOS locally.")
    print(f"Chromium version: {version}")
    print(f"Chromium src: {args.chromium_src}")
    print("Expected disk usage: roughly 100 GB or more.")
    answer = input("Continue? [y/N] ").strip().lower()
    if answer not in ("y", "yes"):
        raise SystemExit("Cancelled.")


def run(
    argv: Sequence[str] | None = None,
    *,
    repo_root: Path | None = None,
    runner: Runner = run_command,
) -> None:
    repo_root = repo_root or repo_root_from_script()
    package_root = browseros_root(repo_root)
    args = parse_args(argv, repo_root)
    args.chromium_src = args.chromium_src.resolve()

    version = read_chromium_version(package_root / "CHROMIUM_VERSION")
    base_commit = (package_root / "BASE_COMMIT").read_text().strip()

    tools = ensure_tools()
    confirm_large_build(args, version)

    print(f"BrowserOS root: {package_root}")
    print(f"Chromium version: {version}")
    print(f"Base commit: {base_commit}")
    print(f"Chromium src: {args.chromium_src}")

    if not args.chromium_src.exists():
        if args.skip_fetch:
            raise SystemExit(f"Chromium source not found: {args.chromium_src}")
        args.chromium_src.parent.mkdir(parents=True, exist_ok=True)
        runner([tools["fetch"], "--no-history", "chromium"], args.chromium_src.parent, None)

    env = os.environ.copy()
    if sys.platform == "win32":
        env["DEPOT_TOOLS_WIN_TOOLCHAIN"] = "0"
    if args.jobs is not None:
        if args.jobs < 1:
            raise SystemExit("--jobs must be greater than 0")
        env["BROWSEROS_NINJA_JOBS"] = str(args.jobs)

    build_command = build_browseros_command(
        chromium_src=args.chromium_src,
        arch=args.arch,
        build_type=args.build_type,
    )
    if args.skip_build:
        build_command.remove("--build")

    runner(build_command, package_root, env)


def main() -> None:
    try:
        run()
    except subprocess.CalledProcessError as exc:
        raise SystemExit(exc.returncode) from exc
    except RuntimeError as exc:
        print(exc, file=sys.stderr)
        raise SystemExit(1) from exc


if __name__ == "__main__":
    main()

```

## bootstrap_build_test.py
```python
import sys
import tempfile
import unittest
from pathlib import Path
from unittest.mock import patch

import bootstrap_build


class BootstrapBuildTest(unittest.TestCase):
    def test_read_chromium_version_from_key_value_file(self):
        with tempfile.TemporaryDirectory() as tmp:
            version_file = Path(tmp) / "CHROMIUM_VERSION"
            version_file.write_text("MAJOR=148\nMINOR=0\nBUILD=7778\nPATCH=97\n")

            self.assertEqual(
                bootstrap_build.read_chromium_version(version_file),
                "148.0.7778.97",
            )

    def test_default_chromium_src_lives_under_browseros_package(self):
        root = Path("repo").resolve()

        self.assertEqual(
            bootstrap_build.default_chromium_src(root),
            root / "packages" / "browseros" / ".chromium" / "src",
        )

    def test_build_command_uses_existing_browseros_pipeline(self):
        command = bootstrap_build.build_browseros_command(
            chromium_src=Path("chromium/src"),
            arch="x64",
            build_type="debug",
        )

        self.assertEqual(
            command,
            [
                sys.executable,
                "-m",
                "build.browseros",
                "build",
                "--chromium-src",
                str(Path("chromium/src")),
                "--arch",
                "x64",
                "--build-type",
                "debug",
                "--setup",
                "--prep",
                "--build",
            ],
        )

    def test_run_fetches_chromium_when_source_is_missing(self):
        with tempfile.TemporaryDirectory() as tmp:
            repo_root = Path(tmp)
            browseros_root = repo_root / "packages" / "browseros"
            browseros_root.mkdir(parents=True)
            (browseros_root / "CHROMIUM_VERSION").write_text(
                "MAJOR=148\nMINOR=0\nBUILD=7778\nPATCH=97\n"
            )
            (browseros_root / "BASE_COMMIT").write_text("abc123\n")
            chromium_src = browseros_root / ".chromium" / "src"
            calls = []

            def record(command, cwd, env=None):
                calls.append((command, cwd, env))

            with patch.object(
                bootstrap_build,
                "resolve_command",
                side_effect=lambda command: f"resolved-{command}",
            ):
                bootstrap_build.run(
                    [
                        "--chromium-src",
                        str(chromium_src),
                        "--yes",
                    ],
                    repo_root=repo_root,
                    runner=record,
                )

            self.assertEqual(calls[0][0], ["resolved-fetch", "--no-history", "chromium"])
            self.assertEqual(calls[0][1], chromium_src.parent)
            self.assertEqual(calls[1][0][:3], [sys.executable, "-m", "build.browseros"])
            self.assertEqual(calls[1][1], browseros_root)


if __name__ == "__main__":
    unittest.main()
```
